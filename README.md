template <typename Any>
Any ArithmeticMean(int size, Any arr[])
{
	Any value = 0;
	for (int i = 0; i < size; i++)
	{
		std::cout << arr[i] << "\t";
		value += arr[i];
	}
	std::cout << "\n\n";
	std::cout << "Среднее арифметическое = " << value / size;
	return value / size;
}
